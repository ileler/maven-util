# maven-util
根据项目中的实际情况写了一个工具类 `com.ileler.maven.util.MavenUtil` 辅助打包、发布等操作。
使用前需要将maven-util.jar放到CLASSPATH下
```xml
					<plugin>
						<artifactId>exec-maven-plugin</artifactId>
						<groupId>org.codehaus.mojo</groupId>
						<executions>
						  <!--clean操作时做些某某某-->
							<execution>
								<id>clean</id>
								<phase>clean</phase>
								<goals>
									 <goal>exec</goal>
								</goals>
								<configuration>  
							        <executable>java</executable>
							        <!--如果maven-util.jar在CLASSPATH路径下，就按以下配置-->
							        <arguments>  
							            <argument>com.ileler.maven.util.MavenUtil</argument>
										<argument>clean</argument>
							            <argument>${virgo.home}</argument>
							        </arguments>
							        <!--如果maven-util.jar在当前路径下，就按以下配置-->
							        <arguments>  
							            <argument>-classpath</argument>
							            <argument>.</argument>
							            <argument>com.ileler.maven.util.MavenUtil</argument>
										<argument>clean</argument>
							            <argument>${virgo.home}</argument>
							        </arguments>
							    </configuration>
							</execution>
							<!--package操作时做些某某某-->
							<execution>
								<id>package</id>
								<phase>package</phase>
								<goals>
									 <goal>exec</goal>
								</goals>
								<configuration>  
							        <executable>java</executable>
							        <arguments>  
							            <argument>com.ileler.maven.util.MavenUtil</argument>
										<argument>deploy</argument>
							            <argument>${basedir}</argument>
							            <argument>${virgo.home}</argument>
							        </arguments>  
							    </configuration>
							</execution>
						</executions>
					</plugin>
					
					<!--打包时从本地其他项目中copy资源到指定路径下-->
					<plugin>
    				<artifactId>exec-maven-plugin</artifactId>
    				<groupId>org.codehaus.mojo</groupId>
    				<executions>
    					<execution>
    						<id>copy-i18n</id>
    						<phase>compile</phase>
    						<goals>
    							 <goal>exec</goal>
    						</goals>
    						<configuration>  
    					        <executable>java</executable>
    					        <arguments>  
    					            <argument>com.ileler.maven.util.MavenUtil</argument>
    					            <argument>copy</argument>
    					            <argument>${basedir}\..\..\org.isli.irms.host\src\main\resources\resources\i18n</argument>
    								<argument>${basedir}/target/classes/WEB-INF/classes/resources/i18n/</argument>
    					        </arguments>  
    					    </configuration>
    					</execution>
    					<execution>
    						<id>copy-template</id>
    						<phase>compile</phase>
    						<goals>
    							 <goal>exec</goal>
    						</goals>
    						<configuration>  
    					        <executable>java</executable>
    					        <arguments>  
    					            <argument>com.ileler.maven.util.MavenUtil</argument>
    					            <argument>copy</argument>
    					            <argument>${basedir}\..\..\org.isli.irms.host\src\main\resources\resources\template</argument>
    					            <argument>${basedir}/target/classes/WEB-INF/template/common</argument>
    					        </arguments>  
    					    </configuration>
    					</execution>
    				</executions>
    			</plugin>
```
